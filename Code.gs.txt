// ====== 初期設定 ======
const FORM_SHEET_NAME = "フォームの回答 1";
const STORAGE_SHEET_NAME = "アプリ保存用";

// 1. はじめに1回だけ「実行」して保存用シートを作成します
function setup() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  if (!ss.getSheetByName(STORAGE_SHEET_NAME)) {
    const sheet = ss.insertSheet(STORAGE_SHEET_NAME);
    sheet.getRange('A1').setValue('{}');
  }
}

// 保存されているデータを読み取り、現行フォーマットに正規化して返す
function _readStoredData_(storageSheet) {
  let rawStr = "{}";
  if (storageSheet) {
    rawStr = storageSheet.getRange('A1').getValue() || "{}";
  }
  let parsed = {};
  try {
    parsed = JSON.parse(rawStr);
  } catch (err) {
    parsed = {};
  }

  // 新フォーマット: {appData, participantsList, sessionsInfo, lastUpdated}
  // 旧フォーマット: appData のみがそのまま入っている
  if (parsed && (parsed.appData !== undefined || parsed.participantsList !== undefined || parsed.sessionsInfo !== undefined)) {
    return {
      appData: parsed.appData || {},
      participantsList: parsed.participantsList || [],
      sessionsInfo: parsed.sessionsInfo || [],
      lastUpdated: parsed.lastUpdated || null
    };
  }
  return {
    appData: parsed || {},
    participantsList: [],
    sessionsInfo: [],
    lastUpdated: null
  };
}

// 2. データの取得（GETリクエスト）
function doGet(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();

  // フォーム回答データの取得
  const formSheet = ss.getSheetByName(FORM_SHEET_NAME);
  let formResponses = [];
  if (formSheet) {
    const data = formSheet.getDataRange().getValues();
    const headers = data[0];
    for (let i = 1; i < data.length; i++) {
      let row = {};
      headers.forEach((h, colIdx) => {
        row[String(h).trim()] = data[i][colIdx];
      });
      formResponses.push(row);
    }
  }

  // 保存済みデータの取得（新/旧フォーマット両対応）
  const storageSheet = ss.getSheetByName(STORAGE_SHEET_NAME);
  const stored = _readStoredData_(storageSheet);

  const response = {
    formResponses: formResponses,
    appData: stored.appData,
    participantsList: stored.participantsList,
    sessionsInfo: stored.sessionsInfo,
    lastUpdated: stored.lastUpdated
  };

  return ContentService.createTextOutput(JSON.stringify(response))
    .setMimeType(ContentService.MimeType.JSON);
}

// 3. データの保存（POSTリクエスト - HTMLフォーム送信対応）
function doPost(e) {
  // 複数端末からの同時書き込み競合を防止
  const lock = LockService.getScriptLock();
  try {
    lock.waitLock(10000);
  } catch (lockErr) {
    return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "他の保存処理が進行中のためタイムアウトしました。もう一度お試しください。" }))
      .setMimeType(ContentService.MimeType.JSON);
  }

  try {
    let payloadStr = '{}';

    // ★ Base64送信（文字化け完全回避）の場合
    if (e.parameter && e.parameter.data_b64) {
      var decoded = Utilities.base64Decode(e.parameter.data_b64);
      payloadStr = Utilities.newBlob(decoded).getDataAsString('UTF-8');
    }
    // フォーム送信（application/x-www-form-urlencoded）の場合（フォールバック）
    else if (e.parameter && e.parameter.data) {
      payloadStr = e.parameter.data;
    }
    // JSON送信（text/plain）の場合
    else if (e.postData && e.postData.contents) {
      payloadStr = e.postData.contents;
    }

    let incoming = {};
    try {
      incoming = JSON.parse(payloadStr);
    } catch (err) {
      incoming = {};
    }

    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const storageSheet = ss.getSheetByName(STORAGE_SHEET_NAME);
    if (!storageSheet) {
      throw new Error("保存用シートが見つかりません。");
    }

    // 既存データ（現在のクラウド状態）を読み込み、送られていないキーは既存値を保持
    const current = _readStoredData_(storageSheet);

    let newAppData, newParticipantsList, newSessionsInfo;
    if (incoming && (incoming.appData !== undefined || incoming.participantsList !== undefined || incoming.sessionsInfo !== undefined)) {
      newAppData = incoming.appData !== undefined ? incoming.appData : current.appData;
      newParticipantsList = incoming.participantsList !== undefined ? incoming.participantsList : current.participantsList;
      newSessionsInfo = incoming.sessionsInfo !== undefined ? incoming.sessionsInfo : current.sessionsInfo;
    } else {
      // 旧クライアント（appDataだけを送ってくる）との互換
      newAppData = incoming || current.appData;
      newParticipantsList = current.participantsList;
      newSessionsInfo = current.sessionsInfo;
    }

    const toStore = {
      appData: newAppData || {},
      participantsList: newParticipantsList || [],
      sessionsInfo: newSessionsInfo || [],
      lastUpdated: new Date().toISOString()
    };

    storageSheet.getRange('A1').setValue(JSON.stringify(toStore));

    return ContentService.createTextOutput(JSON.stringify({ status: "success", lastUpdated: toStore.lastUpdated }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({ status: "error", message: error.message }))
      .setMimeType(ContentService.MimeType.JSON);
  } finally {
    try { lock.releaseLock(); } catch (_) {}
  }
}
